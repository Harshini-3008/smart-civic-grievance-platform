    <script>
        let complaints = [

            {

                type: 'streetlight',
                location: 'Park Road, Secunderabad',
                description: 'Multiple streetlights not functioning',
                status: 'resolved',
                date: '2025-10-08',
                updates: [
                    { date: '2025-10-08', status: 'Complaint Registered', message: 'Assigned to Electrical Department' },
                    { date: '2025-10-09', status: 'Work in Progress', message: 'Technician dispatched to location' },
                    { date: '2025-10-10', status: 'Resolved', message: 'All streetlights have been repaired and are functioning' }
                ]
            }

        let isRecording = false;

        function renderComplaints() {
            const list = document.getElementById('complaintsList');
            list.innerHTML = complaints.map(c => `
                <div class="complaint-item" onclick="viewComplaint('${c.id}')">
                    <div class="complaint-header">
                        <span class="complaint-title">${c.title}</span>
                        <span class="status-badge status-${c.status}">${c.status.toUpperCase()}</span>
                    </div>
                    <div class="complaint-meta">
                        <i class="fas fa-map-marker-alt"></i> ${c.location} • 
                        <i class="fas fa-calendar"></i> ${c.date} • 
                        ID: ${c.id}
                    </div>
                </div>
            `).join('');
        }

        function viewComplaint(id) {
            const complaint = complaints.find(c => c.id === id);
            const modal = document.getElementById('complaintModal');
            const modalTitle = document.getElementById('modalTitle');
            const modalContent = document.getElementById('modalContent');

            modalTitle.textContent = complaint.title;
            modalContent.innerHTML = `
                <p style="margin-bottom: 1rem;"><strong>Complaint ID:</strong> ${complaint.id}</p>
                <p style="margin-bottom: 1rem;"><strong>Type:</strong> ${complaint.type}</p>
                <p style="margin-bottom: 1rem;"><strong>Location:</strong> ${complaint.location}</p>
                <p style="margin-bottom: 1rem;"><strong>Description:</strong> ${complaint.description}</p>
                <p style="margin-bottom: 1rem;"><strong>Status:</strong> <span class="status-badge status-${complaint.status}">${complaint.status.toUpperCase()}</span></p>
                
                <h3 style="margin: 2rem 0 1rem;">Progress Timeline</h3>
                <div class="timeline">
                    ${complaint.updates.map(update => `
                        <div class="timeline-item">
                            <strong>${update.status}</strong>
                            <p style="color: var(--gray); font-size: 0.875rem; margin: 0.25rem 0;">${update.date}</p>
                            <p style="margin-top: 0.5rem;">${update.message}</p>
                        </div>
                    `).join('')}
                </div>
            `;

            modal.classList.add('active');
        }

        function closeModal() {
            document.getElementById('complaintModal').classList.remove('active');
        }

        function getLocation() {
            if (navigator.geolocation) {
                document.getElementById('location').value = 'Detecting location...';
                navigator.geolocation.getCurrentPosition(
                    position => {
                        document.getElementById('location').value = 
                            `Lat: ${position.coords.latitude.toFixed(4)}, Lng: ${position.coords.longitude.toFixed(4)}`;
                    },
                    error => {
                        document.getElementById('location').value = 'Location detection failed. Please enter manually.';
                        document.getElementById('location').readOnly = false;
                    }
                );
            } else {
                alert('Geolocation is not supported by your browser');
            }
        }

        function toggleVoiceInput() {
            const btn = document.getElementById('voiceBtn');
            const text = document.getElementById('voiceText');
            
            if (!isRecording) {
                isRecording = true;
                btn.classList.add('recording');
                text.textContent = 'Recording... (Click to Stop)';
                
                // Simulate voice recording
                setTimeout(() => {
                    if (isRecording) {
                        document.getElementById('description').value = 
                            'There is a large pothole on the main road near the market that has been causing problems for vehicles. It needs immediate attention.';
                        toggleVoiceInput();
                    }
                }, 3000);
            } else {
                isRecording = false;
                btn.classList.remove('recording');
                text.textContent = 'Voice Report (Click to Start)';
            }
        }

        function handleFiles(files) {
            const preview = document.getElementById('filePreview');
            preview.innerHTML = '';
            
            Array.from(files).forEach(file => {
                const reader = new FileReader();
                reader.onload = e => {
                    preview.innerHTML += `
                        <div style="display: inline-block; margin: 0.5rem; position: relative;">
                            <img src="${e.target.result}" style="width: 100px; height: 100px; object-fit: cover; border-radius: 8px;">
                        </div>
                    `;
                };
                reader.readAsDataURL(file);
            });
        }

        document.getElementById('complaintForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const newComplaint = {
                id: 'C' + String(complaints.length + 1).padStart(3, '0'),
                title: document.getElementById('issueType').options[document.getElementById('issueType').selectedIndex].text,
                type: document.getElementById('issueType').value,
                location: document.getElementById('location').value,
                description: document.getElementById('description').value,
                status: 'pending',
                date: new Date().toISOString().split('T')[0],
                updates: [
                    {
                        date: new Date().toISOString().split('T')[0],
                        status: 'Complaint Registered',
                        message: 'Your complaint has been successfully registered and assigned to the relevant department.'
                    }
                ]
            };
            
            complaints.unshift(newComplaint);
            renderComplaints();
            
            // Update stats
            document.getElementById('totalComplaints').textContent = complaints.length;
            document.getElementById('pendingComplaints').textContent = 
                complaints.filter(c => c.status === 'pending').length;
            
            alert('Complaint submitted successfully! Your complaint ID is: ' + newComplaint.id);
            this.reset();
            document.getElementById('filePreview').innerHTML = '';
        });

        // Initialize
        renderComplaints();

                stat.style.transform = 'scale(1.1)';
                setTimeout(() => stat.style.transform = 'scale(1)', 200);
            });
        }, 5000);
